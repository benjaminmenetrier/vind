# VIND bundle for JEDI
cmake_minimum_required( VERSION 3.12 FATAL_ERROR )
find_package( ecbuild 3.6 REQUIRED HINTS ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../ecbuild)
project( vind-bundle VERSION 8.0.0 LANGUAGES C CXX Fortran )
list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/cmake")
include( ecbuild_bundle )

# Default release mode
set( ECBUILD_DEFAULT_BUILD_TYPE Release )

# Options
option( ENABLE_FALFILFA "Compile with FALFILFA" ON )
option( ENABLE_MPI "Compile with MPI" ON )
option( ENABLE_OMP "Compile with OpenMP" ON )
option( ENABLE_CUDA "Compile with CUDA" OFF )
option( ENABLE_MKL "Compile with MKL" OFF )
option( ENABLE_TRANSI "Compile TransI C-interface to trans" ON )
option( ENABLE_ETRANS "Include Limited-Area-Model Transforms" ON )
option( ENABLE_JEDI_MODEL_DATA "Obtain saber/vader test data from jedi-model-data repository (vs tarball)" ON )
option( ENABLE_PYBIND11 "Compile with pybind11" OFF )
option( ENABLE_TORCH "Use Torch" OFF )
option( VENV_DIR "Virtual environment directory" ${CMAKE_CURRENT_BINARY_DIR}/torch_env )
option( ENABLE_IODA_DATA "Obtain ioda test data from ioda-data repository (vs tarball)" ON )
option( ENABLE_UFO_DATA "Obtain ufo test data from ufo-data repository (vs tarball)" ON )
option( BUILD_IODA_CONVERTERS "Build IODA Converters" OFF )

# What to compile?
option( ENABLE_SABER "Compile SABER" ON )
option( ENABLE_OBS "Compile IODA/UFO" ON )
option( ENABLE_CRTM "Compile CRTM" OFF )

# What tests to run?
option( ENABLE_AQ_TESTS "Run aq tests" OFF )
option( ENABLE_GENINT_TESTS "Run genint tests" ON )
option( ENABLE_GLB_TESTS "Run global tests" ON )
option( ENABLE_MONARCH_TESTS "Run regional monarch tests" OFF )
option( ENABLE_MONGLOB_TESTS "Run global monarch tests" OFF )
option( ENABLE_REG_TESTS "Run regional tests" ON )

# Setup
link_directories( ${CMAKE_CURRENT_BINARY_DIR}/lib )
include( GNUInstallDirs )
set( INSTALL_LIB_DIR ${CMAKE_INSTALL_LIBDIR} CACHE STRING "Library installation directory" FORCE )
set( ECBUILD_DEFAULT_BUILD_TYPE Release CACHE STRING "ECBuild default build type" FORCE )
set( CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin )
set( CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib )
set( CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib )
set( CMAKE_INSTALL_RPATH_USE_LINK_PATH ON )
set( CMAKE_BUILD_WITH_INSTALL_RPATH ON )

# Initialize bundle
ecbuild_bundle_initialize()

# Find system Python3 for venv creation
if( NOT Python3_EXECUTABLE )
    find_package( Python3 COMPONENTS Interpreter REQUIRED )
endif()

# Virtual environment
message( STATUS "Virtual environment for Python: " ${VENV_DIR} )
set( VENV_PYTHON ${VENV_DIR}/bin/python )

if( NOT EXISTS ${VENV_PYTHON} )
    # Create virtual environment
    message( STATUS "Creating virtual environment for Python in: " ${VENV_DIR} )
    execute_process(
        COMMAND ${Python3_EXECUTABLE} -m venv ${VENV_DIR}
        RESULT_VARIABLE VENV_RESULT
    )
    if( NOT VENV_RESULT EQUAL 0 )
        message( FATAL_ERROR "Failed to create virtual environment" )
    endif()
endif()

# Reset Python3 to venv
set( ENV{VIRTUAL_ENV} "${VENV_DIR}" )
set( Python3_FIND_VIRTUALENV ONLY )
unset( Python3_EXECUTABLE CACHE )
find_package( Python3 COMPONENTS Interpreter Development REQUIRED )

# Add virtual environment binaries to path
list( APPEND CMAKE_PROGRAM_PATH "${VENV_DIR}/bin" )

# Add site-packages in definitions
file( READ_SYMLINK ${VENV_PYTHON} PYTHON_VERSION )
set( VENV_SITE_PACKAGES "${VENV_DIR}/lib/${PYTHON_VERSION}/site-packages" )

# Use external jedi-cmake or build in bundle
if( DEFINED ENV{jedi_cmake_ROOT} )
  include( $ENV{jedi_cmake_ROOT}/share/jedicmake/Functions/git_functions.cmake )
  list( APPEND CMAKE_MODULE_PATH $ENV{jedi_cmake_ROOT}/share/jedicmake/Modules )
  message( INFO "CMAKE_MODULE_PATH" )
else()
  if (DEFINED ENV{VIND_BUNDLE_SRC_DIR})
    ecbuild_bundle( PROJECT jedicmake SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/jedicmake )
  else()
    ecbuild_bundle( PROJECT jedicmake GIT "https://github.com/jcsda/jedi-cmake.git" TAG 40d521f9a2d796fcbc6234d77abceeffefb8eb7f RECURSIVE )
  endif()
  include( jedicmake/cmake/Functions/git_functions.cmake )
  list( APPEND CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/jedicmake/CMakeModules/Modules )
endif()

if (DEFINED ENV{VIND_BUNDLE_SRC_DIR})
  if( ENABLE_PYBIND11 )
    # pybind11
    set( PYBIND11_DIR $ENV{VIND_BUNDLE_SRC_DIR}/pybind11 )
  endif()

  # ECMWF tools
  ecbuild_bundle( PROJECT eckit SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/eckit )
  ecbuild_bundle( PROJECT fckit SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/fckit )
  ecbuild_bundle( PROJECT fiat SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/fiat )
  ecbuild_bundle( PROJECT ectrans SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/ectrans )
  ecbuild_bundle( PROJECT atlas SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/atlas )

  if( ENABLE_FALFILFA )
    # eccodes and falfilfa
    set( ENABLE_AEC OFF )
    ecbuild_bundle( PROJECT eccodes SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/eccodes )
    ecbuild_bundle( PROJECT falfilfa SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/falfilfa )
  endif()

  # oops
  ecbuild_bundle( PROJECT oops SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/oops )

  # jedi-data
  ecbuild_bundle( PROJECT jedi-model-data SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/jedi-model-data )

  # vader
  ecbuild_bundle( PROJECT vader SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/vader )

  if( ENABLE_SABER )
    # gsibec
    ecbuild_bundle( PROJECT gsibec SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/gsibec )

    # saber
    ecbuild_bundle( PROJECT saber SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/saber )
  endif()

  if( ENABLE_CRTM )
    # crtm
    ecbuild_bundle( PROJECT crtm SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/crtm )
  endif()

  if( ENABLE_OBS )
    # Install pycodestyle
    execute_process(
        COMMAND ${VENV_DIR}/bin/pip install pycodestyle
        RESULT_VARIABLE PIP_RESULT
        OUTPUT_VARIABLE PIP_OUTPUT
    )
    if( NOT PIP_RESULT EQUAL 0 )
        message( STATUS "pycodestyle installation output: " ${OUTPUT_VARIABLE PIP_OUTPUT} )
        message( FATAL_ERROR "Failed to install pycodestyle in virtual environment" )
    endif()

    # ioda
    ecbuild_bundle( PROJECT ioda SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/ioda )
    if( BUILD_IODA_CONVERTERS )
        ecbuild_bundle( PROJECT iodaconv GIT SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/iodaconv )
    endif()

    # ufo
    ecbuild_bundle( PROJECT ufo SOURCE $ENV{VIND_BUNDLE_SRC_DIR}/ufo )
  endif()
else()
  if( ENABLE_PYBIND11 )
    # pybind11
    ecbuild_git( PROJECT pybind11 DIR ${CMAKE_CURRENT_SOURCE_DIR}/pybind11 URL https://github.com/pybind/pybind11.git BRANCH master )
    set( PYBIND11_DIR bundle/pybind11 )
  endif()

  # ECMWF tools
  ecbuild_bundle( PROJECT eckit GIT "https://github.com/ecmwf/eckit.git" TAG 1.32.2 )
  ecbuild_bundle( PROJECT fckit GIT "https://github.com/ecmwf/fckit.git" TAG 0.14.0 )
  ecbuild_bundle( PROJECT fiat GIT "https://github.com/ecmwf-ifs/fiat.git" TAG 1.6.1 )
  ecbuild_bundle( PROJECT ectrans GIT "https://github.com/ecmwf-ifs/ectrans.git" TAG 3094b24ec9c1862772e2b02f4e06c797247d57fc )
  ecbuild_bundle( PROJECT atlas GIT "https://github.com/ecmwf/atlas.git" TAG 0.44.0 )

  if( ENABLE_FALFILFA )
    # eccodes and falfilfa
    set( ENABLE_AEC OFF )
    ecbuild_bundle( PROJECT eccodes GIT "https://github.com/ecmwf/eccodes.git" TAG 2.44.0 )
    ecbuild_bundle( PROJECT falfilfa GIT "https://github.com/benjaminmenetrier/falfilfa.git" TAG 5767282763f5a937cabee61596f751bc044b01d7 )
  endif()

  # oops
  ecbuild_bundle( PROJECT oops GIT "https://github.com/jcsda/oops.git" TAG 344ee6312e4e4e4dfb8660df3683331177883047 )

  # jedi-data
  ecbuild_bundle( PROJECT jedi-model-data GIT "https://github.com/jcsda-internal/jedi-model-data.git" TAG 12e2f5c657757c8305330c99868df070960e7d0d )
  
  # vader
  ecbuild_bundle( PROJECT vader GIT "https://github.com/jcsda/vader.git" TAG c4c75e43c54bf8a4edcd83ad6b6b919da1ce58dd )

  if( ENABLE_SABER )
    # gsibec
    ecbuild_bundle( PROJECT gsibec GIT "https://github.com/geos-esm/gsibec.git" TAG c9fdaf5eeb8d37bd918f6542af7a413bb343aef9 )

    # saber
    ecbuild_bundle( PROJECT saber GIT "https://github.com/benjaminmenetrier/saber.git" TAG a51261f65d73382dee055bdace6330c82ed90637 )
  endif()
  
  if( ENABLE_CRTM )
    # crtm
    ecbuild_bundle( PROJECT crtm GIT "https://github.com/jcsda/crtmv3.git" TAG 82c746009c27585e5cd323365286d6f57d65db0d )
  endif()

  if( ENABLE_OBS )
    # ioda-data
    ecbuild_bundle( PROJECT ioda-data GIT "https://github.com/jcsda-internal/ioda-data.git" TAG 04b68d277682c121a536ebff75cbdc954cbb129d )

    # ioda
    ecbuild_bundle( PROJECT ioda GIT "https://github.com/jcsda/ioda.git" TAG aa392a3fc2db3bfe98459e1848cbec0720cd589c )

    # ioda-converters
    if( BUILD_IODA_CONVERTERS )
      ecbuild_bundle( PROJECT iodaconv GIT "https://github.com/jcsda/ioda-converters.git" TAG ed3ec4131fed7d7b8e6a3e1b8dcf2be639bd478d )
    endif()

    # ufo-data
    ecbuild_bundle( PROJECT ufo-data GIT "https://github.com/jcsda-internal/ufo-data.git" TAG d86d921cc67f91453ce6d173c2edac9ca02bbd16 )

    # ufo
    ecbuild_bundle( PROJECT ufo GIT "https://github.com/jcsda/ufo.git" TAG 056bc01dd7e0467c632e6601f2d48e6237f6d11b )
  endif()
endif()

# vind
ecbuild_bundle( PROJECT vind SOURCE ..)

ecbuild_bundle_finalize()
